################################################################
## Post-compilation for all targets
################################################################

IF(NOT WIN32)
  ADD_DEFINITIONS("-O2 -Wall")
ENDIF(NOT WIN32)

################################################################
## dijkstra()
################################################################

ADD_MEX_FILE(dijkstra
  dijkstra.cpp)

################################################################
## installation of targets
################################################################

IF(WIN32)
  INSTALL(TARGETS
    dijkstra
    RUNTIME
    DESTINATION ${CMAKE_CURRENT_SOURCE_DIR})
ELSE(WIN32)
  INSTALL(TARGETS
    dijkstra
    LIBRARY
    DESTINATION ${CMAKE_CURRENT_SOURCE_DIR})
ENDIF(WIN32)
